#version 330 core

layout(location = 0) in vec3 inPos;
layout(location = 1) in vec3 inNormal;

out vec3 vFragPos;
out vec3 vNormal;

uniform mat4 vuModel;
uniform mat4 vuView;
uniform mat4 vuPerspective;

void main(){
	gl_Position = vuPerspective * vuView * vuModel * vec4(inPos, 1.0);
	vFragPos = vec3(vuModel * vec4(inPos, 1.0));
	vNormal = mat3(transpose(inverse(vuModel))) * inNormal;
}
