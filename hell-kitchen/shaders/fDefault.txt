#version 330 core

in vec3 vFragPos;
in vec2 vTex;

out vec4 fColor;

uniform sampler2D tex;

float near = 0.1;
float far = 100.0;

float linearizeDepth(float z){
	float ndc = z * 2 - 1.0;
	return (2.0 * near * far) / (far + near - z * (far - near));
}


void main(){	
	//fColor = texture(tex, vTex);
	float depth = linearizeDepth(gl_FragCoord.z) / far;
	fColor = vec4(vec3(depth), 1.0);
}
