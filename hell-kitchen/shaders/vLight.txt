#version 330 core

layout(location = 0) in vec3 inPos;
layout(location = 1) in vec3 inNormal;
layout(location = 2) in vec2 inTex;

out vec3 vFragPos;
out vec3 vNormal;
out vec2 vTex;

uniform mat4 vuModel;
uniform mat4 vuView;
uniform mat4 vuProjection;

void main(){
	vFragPos = inPos;
	vNormal = inNormal;
	vTex = inTex;
	gl_Position = vuProjection * vuView * vuModel * vec4(inPos, 1.0);
}
